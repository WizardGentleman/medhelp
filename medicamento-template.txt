# TEMPLATE DE DESIGN PARA PÁGINAS DE MEDICAMENTOS
# Baseado em app\medicamentos\metoclopramida.tsx

## ESTRUTURA BÁSICA DO COMPONENTE:

```tsx
import { ScreenHeader } from '@/components/ScreenHeader';
import { theme } from '@/styles/theme';
import { ScrollView, StyleSheet, Text, View, TouchableOpacity } from 'react-native';
import React, { useState } from 'react';

export default function [NomeMedicamento]Screen() {
  const [expanded, setExpanded] = useState([
    false, false, false, false, false, false, false // Ajustar quantidade conforme seções
  ]);
  
  const toggle = (idx: number) => {
    setExpanded(prev => prev.map((v, i) => i === idx ? !v : v));
  };
  
  return (
    <View style={styles.container}>
      <ScreenHeader title="[Nome do Medicamento (Nome Comercial)]" />
      <ScrollView style={styles.content}>
        {/* Seções do medicamento */}
      </ScrollView>
    </View>
  );
}
```

## ESTRUTURA DE CADA SEÇÃO:

```tsx
{/* [NÚMERO]. [TÍTULO DA SEÇÃO] */}
<TouchableOpacity
  activeOpacity={0.9}
  style={[styles.sectionCard, styles.card[NomeSeção]]}
  onPress={() => toggle([índice])}
>
  <Text style={[styles.sectionTitle, styles.title[NomeSeção]]}>[NÚMERO]. [TÍTULO DA SEÇÃO]:</Text>
  {expanded[[índice]] && (
    <>
      <Text style={styles.paragraph}>
        • [Conteúdo principal]
      </Text>
      <Text style={styles.subParagraph}>
        ◦ [Conteúdo secundário/detalhes]
      </Text>
    </>
  )}
</TouchableOpacity>
```

## DICAS DE FORMATAÇÃO:

- Use • para pontos principais
- Use ◦ para subpontos (com marginLeft)
- Para destacar avisos importantes, use:
  ```tsx
  <Text style={[styles.paragraph, { color: '#388E3C', fontWeight: 'bold' }]}>
    • AVISO IMPORTANTE
  </Text>
  ```
- Mantenha consistência nos títulos das seções (sempre em MAIÚSCULAS)
- Use dois pontos (:) após cada título de seção
- Organize o conteúdo de forma hierárquica com paragraph e subParagraph

## COMPORTAMENTO INTERATIVO:

- TouchableOpacity com activeOpacity={0.9} para feedback visual suave
- Sistema de toggle para expandir/colapsar seções
- Iniciar com todas as seções colapsadas (false no estado)
- ScrollView para permitir navegação em conteúdo extenso
